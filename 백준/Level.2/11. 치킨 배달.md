# 치킨 배달

### [ 문제 정보 ]
> **백준**: https://www.acmicpc.net/problem/15686
> 
> **난이도**: 2
>
> **유형**: #완전탐색


### [ 문제 풀이 ]
```C++
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int n, m, ret = 987654321;
int arr[51][51];
vector<pair<int, int>> home, chicken;
vector<vector<int>> selected;

void combi(int start, vector<int> b) {
    if (b.size() == m) {
        selected.push_back(b);
        return;
    }

    for (int i = start + 1; i < chicken.size(); i++) {
        b.push_back(i);
        combi(i, b);
        b.pop_back();
    }

    return;
}

int solve(pair<int, int> h, int s) {
    return abs(chicken[s].first - h.first) + abs(chicken[s].second - h.second);
}
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);

    cin >> n >> m;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> arr[i][j];
            if (arr[i][j] == 1) {
                home.push_back({i, j});
            }
            else if (arr[i][j] == 2) {
                chicken.push_back({i, j});
            }
        }
    }

    combi(-1, {});

    for (vector<int> sel : selected) {
        int cal = 0;
        for (pair<int, int> h : home) {
            int d = 987654321;
            for (int s : sel) {
                d = min(d, solve(h, s));
            }
            cal += d;
        }
        ret = min(cal, ret);
    }

    cout << ret << '\n';

    return 0;
}
```
