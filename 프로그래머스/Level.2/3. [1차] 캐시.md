### [ 문제 풀이 ]
```C++
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

vector<string> v;
string change(string s) {
    for(int i = 0; i < s.size(); i++) {
        if(s[i] >= 'A' && s[i] <= 'Z') {
            s[i] = s[i] - ('A' - 'a');
        }
    }
    return s;
}
int solution(int cacheSize, vector<string> cities) {
    int answer = 0;
    for(int i = 0; i < cities.size(); i++) {
        cities[i] = change(cities[i]);
    }
    for(int i = 0; i < cities.size(); i++) {
        bool hit = false;
        for(int j = 0; j < v.size(); j++) {
            if(v[j] == cities[i]) {
                hit = true;
                v.erase(v.begin() + j);
                v.push_back(cities[i]);
                break;
            }
        }
        if(hit) {
            answer++;
        }
        else {
            if(cacheSize) {
                if(cacheSize == v.size()) {
                    v.erase(v.begin());
                }
                v.push_back(cities[i]);
            }
            answer += 5;
        }
    }
    return answer;
}
```
